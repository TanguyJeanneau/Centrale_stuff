In [8]:
x = np.linspace(-np.pi, np.pi, 100)

def signal(n, t):
    l = [4/(np.pi*(2*k+1)) * np.sin((2*k+1)*np.pi*t) for k in range(0, n)]
    return sum(l)

for i in range(6):
    n = 2*i+1
    y = [signal(n, t) for t in x]
    plt.subplot(3, 2, i+1)
    plt.plot(x, y)
    plt.title('$n = {}$'.format(n))
    plt.xticks([])
    plt.yticks([])
    
plt.tight_layout()

def mean(l):
    return sum(l)/len(l)

def median(l):
    tmp = l.copy()
    tmp.sort()
    return tmp[np.int(len(l)/2)]

def var(l):
    L = np.array(l)
    return ((L - mean(L)) ** 2).sum()/len(L)

def std(l):
    return np.sqrt(var(l))
